CREATE TABLE Brand(
id INT IDENTITY PRIMARY KEY,
ten_brand NVARCHAR(200),
ten_viet_tat NVARCHAR(50),
img NVARCHAR(250),
user_them NVARCHAR(50),
user_update NVARCHAR(50),
ngay_them DATETIME,
ngay_update DATETIME,
stt NVARCHAR(100),
active BIT
)


CREATE TABLE District(
id INT IDENTITY PRIMARY KEY,
district_name NVARCHAR(50),
ten_viet_tat NVARCHAR(50),
stt NVARCHAR(100),
active BIT
)


CREATE TABLE Regions(
id INT IDENTITY PRIMARY KEY,
name NVARCHAR(50),
ten_viet_tat NVARCHAR(50),
khu_vuc NVARCHAR(100),
vung_mien NVARCHAR(50),
active BIT
)



CREATE TABLE Restaurants(
id INT IDENTITY PRIMARY KEY,
name NVARCHAR(100),
brand NVARCHAR(20),
address NVARCHAR(250),
district NVARCHAR(20),
sdt NVARCHAR(20),
ten_viet_tat NVARCHAR(50),
mail_res NVARCHAR(100),
Regions_name NVARCHAR(50),
stt NVARCHAR(100),
active BIT
)


CREATE TABLE Newpost(
id INT IDENTITY PRIMARY KEY,
restaurant_id NVARCHAR(100),
brand_id NVARCHAR(100),
district_id NVARCHAR(50),
region_id NVARCHAR(50),
note NVARCHAR(250),
time_from time,
time_to time,
staff_quantity NVARCHAR(50),
user_dang_bai NVARCHAR(50),
active BIT
)
